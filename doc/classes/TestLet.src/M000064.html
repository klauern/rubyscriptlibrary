  def test_assignment
    with Let do
      let :x => 0, :y => 1 do
        assert_equal(1, x + y)
        let :x => 2 do
          assert_equal(3, x + y) 
          x = 5
          assert_equal(5, x)
          assert_equal(2, x()) # assignment creates a local that shadows the method.
          self.x = 7
          assert_equal(5, x)
          assert_equal(7, x()) # you can hack around the read only nature, but it's not pretty
        end
        assert_equal(0, x)
      end
    end
  end
