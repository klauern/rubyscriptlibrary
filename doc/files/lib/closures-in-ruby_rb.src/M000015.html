def call_with_too_many_args(closure)
    begin
        puts "closure arity: #{closure.arity}"
        closure.call(1,2,3,4,5,6)
        puts "Too many args worked"
    rescue Exception => e
        puts "Too many args threw exception #{e.class}: #{e}"
    end
end
